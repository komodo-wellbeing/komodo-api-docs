Meta-information for the cubes defined in the data model can be retrieved.

## Request

<ApiRequest
    type="GET"
    url="/cube/v1/meta"
    request={{ headers: [{ name: 'Authorization', description: 'Token returned by authentication process' }] }}
    responses={[
        {
            status: 200,
            example: `{
    "cubes": [
        "public": true,
        "name": "Groups",
        "title": "Groups",
        "dimensions": [
            {
                "name": "Groups.id",
                "title": "Groups Id",
                "description": "The group's ID in Komodo Wellbeing",
                "type": "number",
                "shortTitle": "Id",
                "suggestFilterValues": true,
                "isVisible": true,
            },
            {
                "name": "Groups.name",
                "title": "Groups Name",
                "description": "The name of the group",
                "type": "string",
                "shortTitle": "Name",
                "suggestFilterValues": true,
                "isVisible": true,
            },
            ...
        ],
        "measures": [
            {
                "name": "Groups.count",
                "title": "Groups Count",
                "shortTitle": "Count",
                "type": "number",
                "aggType": "count",
                "isVisible": true,
            },
            ...
        ],
        "segments": [...],
        ...
    ]
}`,
        },
        {
            status: 401,
            description: 'Missing or invalid authentication token',
            example: `{
    "error": "Unauthorized",
}`,
        },
    ]}
/>
